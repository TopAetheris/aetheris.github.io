本文阐述如何创建免费静态站点


# 登录 GitHub 后，点击页面右上角的 "+" 号，选择 "New repository"。
# 在 "Repository name" 中输入你的仓库名称
# 在 "Description" 中可以输入仓库的描述（可选）。
# 重要： 在 "Public" 和 "Private" 选项中，选择 "Private"，将仓库设置为私有。
# 点击 "Create repository" 完成仓库创建。
# 点击 "Add file" 按钮，选择 "Create new file"。在文件名中输入 "index.html"。
# 进入仓库的 "Settings" 页面，找到 "Pages" 部分。
# 在 "Branch" 中选择你的主分支（通常是 main 或 master），然后点击 "Save"。
# 点击 "Add file" 按钮，选择 "Create new file" 在文件内容区域输入你的 License 文本。你可以选择一个适合你的开源许可证

Webpack 配置示例


const path = require('path');
const HtmlWebpackPlugin = require('html-webpack-plugin');

module.exports = {
  entry: './src/index.js', 
  output: {
    filename: '[name].[contenthash].js',
    path: path.resolve(__dirname, 'dist'),
  },
  module: {
    rules: [
      {
        test: /\.css$/,
        use: ['style-loader', 'css-loader'],
      },
      {
        test: /\.(jpg|png|gif|svg)$/,
        use: ['file-loader'],
      },
    ],
  },
  plugins: [
    new HtmlWebpackPlugin({
      template: './index.html',
    }),
  ],
  mode: 'production',
};
